<script type="application/lua">
    local js = require "js"
    local window = js.global
    local document = window.document
    local element = document:getElementById("axum-february")
    local btn = document:getElementById("increase-btn")
    local dec_btn = document:getElementById("decrease-btn")
    local counter = 0
    local function tick()
        element.innerHTML = "axum-february::" .. counter
        counter = counter + 1
    end
    btn:addEventListener("click", function()
        counter = counter + 20
        element.innerHTML = "axum-february::" .. counter
    end)
    dec_btn:addEventListener("click", function()
        if counter >= 20 then
            counter = counter - 20
            element.innerHTML = "axum-february::" .. counter
        end
    end)
    local grid_label = document:getElementById("grid-id-label")
    local last_clicked_btn = nil
    for col = 0, 2 do
        for row = 0, 3 do
            local id = "btn-grid-" .. col .. "-" .. row
            local b = document:getElementById(id)
            if b then
                b:addEventListener("click", function()
                    if last_clicked_btn then
                        last_clicked_btn.innerHTML = "N"
                    end
                    b.innerHTML = "+"
                    last_clicked_btn = b
                    grid_label.innerHTML = "ID: " .. id
                end)
            end
        end
    end

    tick()
    window:setInterval(tick, 100)
</script>
